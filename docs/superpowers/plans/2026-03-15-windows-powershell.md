Create `internal/perl/shell_templates/pvm.ps1`:

```powershell
# PVM Shell Integration for PowerShell
# Template file - Generated by PVM init

# Find PVM executable dynamically
function _pvm_find_executable {
    # First, try to find pvm in PATH
    $pvmCmd = Get-Command pvm -ErrorAction SilentlyContinue
    if ($pvmCmd) {
        return $pvmCmd.Source
    }

    # Fallback to the path that was used during generation
    $fallbackPath = "{{.PVMPath}}"
    if (Test-Path $fallbackPath) {
        return $fallbackPath
    }

    Write-Error "PVM executable not found. Please ensure PVM is installed and in PATH."
    return $null
}

# Cache PVM executable path
$script:PVM_EXEC_CACHE = $null

function _pvm_executable {
    if (-not $script:PVM_EXEC_CACHE) {
        $script:PVM_EXEC_CACHE = _pvm_find_executable
        if (-not $script:PVM_EXEC_CACHE) {
            return $null
        }
    }
    return $script:PVM_EXEC_CACHE
}

# Update PATH with current Perl version
function _pvm_update_perl_path {
    $pvmExe = _pvm_executable
    if (-not $pvmExe) { return }

    $dataHome = if ($env:XDG_DATA_HOME) { $env:XDG_DATA_HOME } else { Join-Path $env:LOCALAPPDATA "pvm" }
    $binHome = if ($env:XDG_BIN_HOME) { $env:XDG_BIN_HOME } else { Join-Path $env:LOCALAPPDATA "pvm" "bin" }

    # Clean existing PVM perl paths from PATH
    $pathParts = $env:PATH -split ';' | Where-Object {
        $_ -and ($_ -notmatch '[/\\]pvm[/\\]versions[/\\][^/\\]+[/\\]bin$')
    }

    # Ensure bin home is at the front
    if ($binHome -notin $pathParts) {
        $pathParts = @($binHome) + $pathParts
    }

    # Set clean PATH temporarily
    $env:PATH = $pathParts -join ';'

    # Ask PVM what version should be active
    $currentVersion = & $pvmExe current --bare 2>$null
    if ($currentVersion -and $currentVersion -ne 'system') {
        $perlBin = Join-Path $dataHome "versions" $currentVersion "bin"
        if (Test-Path $perlBin) {
            $env:PATH = "$perlBin;$env:PATH"
        }
    }

    # Handle library environment if PVM_PERL_LIBRARY is set
    if ($env:PVM_PERL_LIBRARY) {
        $libraryDir = Join-Path $dataHome "environments" $env:PVM_PERL_LIBRARY
        if (Test-Path $libraryDir) {
            $libraryLib = Join-Path $libraryDir "lib" "perl5"
            if (Test-Path $libraryLib) {
                if ($env:PERL5LIB) {
                    $env:PERL5LIB = "$libraryLib;$env:PERL5LIB"
                } else {
                    $env:PERL5LIB = $libraryLib
                }
            }
            $libraryBin = Join-Path $libraryDir "bin"
            if (Test-Path $libraryBin) {
                $env:PATH = "$libraryBin;$env:PATH"
            }
        }
    } elseif ($env:PERL5LIB) {
        # Clear PVM environment paths from PERL5LIB
        $env:PERL5LIB = ($env:PERL5LIB -split ';' | Where-Object {
            $_ -notmatch '[/\\]pvm[/\\]environments[/\\][^/\\]+[/\\]lib[/\\]perl5$'
        }) -join ';'
    }
}

# Initialize PVM
function _pvm_init {
    # Add bin home to PATH
    $binHome = if ($env:XDG_BIN_HOME) { $env:XDG_BIN_HOME } else { Join-Path $env:LOCALAPPDATA "pvm" "bin" }
    if ($env:PATH -notlike "*$binHome*") {
        $env:PATH = "$binHome;$env:PATH"
    }

    # Update PATH with current Perl version
    _pvm_update_perl_path
}

# Main pvm function that intercepts version-switching commands
function global:pvm {
    $pvmExe = _pvm_executable
    if (-not $pvmExe) { return }

    $command = $args[0]
    if ($command -eq 'use') {
        # Handle 'pvm use' with shell integration
        $version = $args[1]
        if (-not $version) {
            Write-Host "Usage: pvm use <version>"
            return
        }
        # Run sh-use and execute the output
        $shellCode = & $pvmExe sh-use $version 2>&1
        # Parse PowerShell-style env assignments from output
        foreach ($line in $shellCode) {
            if ($line -match '^\$env:(\w+)\s*=\s*(.+)$') {
                Set-Item "Env:$($Matches[1])" $Matches[2].Trim("'`"")
            } elseif ($line -match '^Remove-Item') {
                Invoke-Expression $line
            } elseif ($line -match '^Write-Host') {
                Invoke-Expression $line
            }
        }
        _pvm_update_perl_path
    } elseif ($command -eq 'env' -and $args[1] -eq 'activate') {
        # Handle 'pvm env activate' with shell integration
        $envName = $args[2]
        if (-not $envName) {
            Write-Host "Usage: pvm env activate <name>"
            return
        }
        $shellCode = & $pvmExe sh-env-activate $envName 2>&1
        foreach ($line in $shellCode) {
            if ($line -match '^\$env:(\w+)\s*=\s*(.+)$') {
                Set-Item "Env:$($Matches[1])" $Matches[2].Trim("'`"")
            }
        }
        _pvm_update_perl_path
    } else {
        # Delegate all other commands to the pvm binary
        & $pvmExe @args
    }
}

# Initialize
_pvm_init

# Set up completion
if ($env:PVM_SKIP_NETWORK_CALLS -ne '1') {
    $pvmExe = _pvm_executable
    if ($pvmExe) {
        & $pvmExe completion powershell 2>$null | Out-String | Invoke-Expression
    }
}

# Output fortune quote
Write-Host "{{.FortuneQuote}}"
```

```go
func DetectShell() (ShellType, error) {
    // Check for PowerShell on any OS (pwsh runs on Linux/macOS too)
    if os.Getenv("PSModulePath") != "" {
        return ShellPowerShell, nil
    }

    // Check SHELL env var (set on Unix, sometimes on Windows via Git Bash)
    shellPath := os.Getenv("SHELL")
    if shellPath != "" {
        shellName := filepath.Base(shellPath)
        switch shellName {
        case "bash":
            return ShellBash, nil
        case "zsh":
            return ShellZsh, nil
        case "fish":
            return ShellFish, nil
        }
    }

    // OS-based fallback
    if runtime.GOOS == "windows" {
        return ShellCmd, nil
    }
    return ShellBash, nil
}
```
